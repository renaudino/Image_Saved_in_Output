1function [Eout, Rt, Rh] = RefFieldUP (pblo, blo, E_re, R_re, qr_re, n_re, Ein)
  2
  3% Calculates the reflected field
  4% blo : block number for which the reflected field is evaluated
  5% E : transmitter position
  6% R : receiver position
  7% qr : reflection point
  8% n : ext. norm.
  9% Ein : incident field
 10% Eout : field at the observation point (R)
 11
 12% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 13% History: 2011.9.26
 14
 15% Updated by: Xuhong Li <xuhong@ftw.at>
 16% History: 2013.08.01
 17%          2014.02.06(save the reflection coefficients)
 18
 19coder.extrinsic('keyboard');
 20
 21% global CEP eta GCEP Geta GRcoeff K Rcoeff vnfaces wedges;
 22global CEP eta K Rcoeff;
 23
 24%%%%%%%%% Incident wave coordinate system calculation : eti & ehi %%%%%%%%%
 25% eti is the perpendicular vector to the incident plane
 26% ehi is the parallel vector to the incident plane
 27
 28vr                  = zeros(3,1);
 29
 30E                   = reshape(E_re,3,1);
 31qr                  = reshape(qr_re,3,1);
 32n                   = reshape(n_re,3,1);
 33R                   = reshape(R_re,3,1);
 34
 35vi                  = qr-E;
 36di                  = norm(qr-E);
 37vi                  = 1/di*vi;
 38
 39vr(:,1)             = R-qr;
 40dr                  = norm(R-qr);
 41vr                  = 1/dr*vr;
 42
 43eti                 = cross(vi,n);
 44
 45csfi2               = complex(norm(eti)^2,0);
 46ccfi                = complex(max(0,real(sqrt(1-csfi2))),0);
 47
 48if (norm(eti) <= 1.0e-6) % if incident wave is normal to the surface, i.e. (cross-product = 0) <=> (theta_inc = 0)
 49    
 50    fprintf('   test -> RefFieldUP \n');
 51    keyboard;
 52    
 53    [eti,ehi]       = cart2rayfixed(vi); % (OR): eti -> unitary parallel vector
 54                                         %       ehi -> unitary perpendicular vector
 55    
 56% %     eti    = [0 0 1].';
 57%     
 58%     d          = zeros(1,6);
 59%     for i = 1:6
 60%         d(1,i) = sum(n_re-squeeze(vnfaces(blo,i,:)));
 61%     end
 62%     [~,ind]     = min(abs(d));
 63%     
 64%     P1          = reshape(wedges(blo,ind,1,:),3,1);
 65%     P2          = reshape(wedges(blo,ind,2,:),3,1);
 66%     P4          = reshape(wedges(blo,ind,4,:),3,1);
 67%     
 68%     if abs(P1(1)-P2(1)) > abs(P1(1)-P4(1))
 69%         P       = [P1 P2];
 70%     else
 71%         P       = [P1 P4];
 72%     end
 73%     [~,ind]     = max(P(1,:));
 74%     
 75%     if (acosd(dot(vi,n)) < 90) % incident wave and normal vector to the surface have same direction
 76%         eti     = P(:,ind)-P(:,setdiff(1:2,ind));
 77%     else % incident wave and normal vector to the surface have opposite direction
 78%         eti     = P(:,setdiff(1:2,ind))-P(:,ind);
 79%     end
 80else
 81    eti             = 1/norm(eti)*eti;
 82    ehi             = cross(eti,vi);
 83end
 84
 85ehr                 = cross(eti,vr);
 86
 87%%%%%%%%%%%%%%%%% E components in the SW-fixed coordinate system %%%%%%%%%%%%%%%
 88coEh                = ehi'*Ein;
 89coEt                = eti'*Ein;
 90
 91if (CEP(blo) == 1)
 92    Rt              = complex(-1,0); % -> perpendicular polarization coefficient
 93    Rh              = complex(1,0); % -> parallel polarization coefficient
 94else
 95    if (pblo == -1)
 96        Ct          = eta(blo)-csfi2;
 97        Ch          = Ct/eta(blo)^2;
 98    else
 99        Ct          = (eta(blo)/eta(pblo))-csfi2;
100        Ch          = Ct/(eta(blo)/eta(pblo))^2;
101    end
102    Rt              = (ccfi-sqrt(Ct))/(ccfi+sqrt(Ct));
103    Rh              = (ccfi-sqrt(Ch))/(ccfi+sqrt(Ch));
104end
105
106fac                 = exp(-1i*K*dr);
107
108Eout                = fac*((ehr*(Rh*coEh))+(eti*(Rt*coEt)));
109Eout                = Rcoeff(blo)*Eout; % updated on 18.10.2011: multiplying the field and the R coefficient that  

110                                        %                        takes into account the diffuse scattering power reduction
111
