1function Falphr = Falpha(alpha,cost)
 2% F_alpha function
 3% Calculates the F_alphar (refer to the paper Vittorio D. 2007)
 4
 5% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 6% History:  2011.10.3
 7sint = sqrt(1-cost.^2);
 8dav = 1/(2.^alpha);
 9sumj = 0;
10for j = 0:alpha
11    alphabin = nchoosek(alpha,j);   % binomial coefficent
12    dpi = 2*pi/(j+1);
13    if (mod(j,2) == 0)
14        Ij = dpi;
15    else 
16        sumw = 0;
17        for w = 0:((j-1)/2)
18            omegabin = nchoosek(2*w,w);
19            sratio = (sint.^(2*w))/(2.^(2*w));
20            sumw = sumw+(omegabin*sratio);
21        end 
22        Ij = dpi*cost*sumw;
23    end 
24    sumj = sumj+(alphabin*Ij);
25end
26Falphr = sumj*dav;
