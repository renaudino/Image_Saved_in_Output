  407     static const int32_T iv1[2] = { 1, 24 };

  410       '-', '>', ' ', 'R', 'e', 'f', 'F', 'i', 'e', 'l', 'd', 'U', 'P', ' ', '\\',
  411       'n' };
  412   
  413     real_T dv0[9];

  416   
  417     creal_T Ch;

  425     real_T dv3[3];

  431     emlrtStack b_st;
  432     emlrtStack c_st;
  433     emlrtStack d_st;
  434     st.prev = sp;
  435     st.tls = sp->tls;
  436     b_st.prev = &st;
  437     b_st.tls = st.tls;
  438     c_st.prev = &b_st;
  439     c_st.tls = b_st.tls;
  440     d_st.prev = &c_st;
  441     d_st.tls = c_st.tls;
  442   
  443     /*  Diffuse scattering field */
  444     /*  Refer to the paper Vittorio D. 2007 */
  445     /*  start : first point */
  446     /*  fin : last point */
  447     /*  sf: scattering face  */
  448     /*  sp: scattering point */
  449     /*  surf: size of element */
  450     /*  Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at> */
  451     /*  History:  2011.10.3 */
  452     /*  Updated by: Xuhong Li <xuhong@ftw.at> */
  453     /*  History: 2013.08.01 */
  454     /*  global vnfaces Model Ar Lam Aii GAr GLam GAii Scoeff kxpol GModel GScoeff; */
  455     num = 0.0;
  456     denom = 0.0;
  457     for (i = 0; i < 3; i++) {
  458       fin[i] = fin_re[i];
  459       si[i] = sp_re[i] - start_re[i];
  460       start[i] = start_re[i];
  461       b_sp[i] = sp_re[i];
  462     }
  463   
  464     *ri = norm(si);
  465     y = 1.0 / *ri;
  466     for (i1 = 0; i1 < 3; i1++) {
  467       si[i1] *= y;
  468     }
  469   
  470     /*  incidence unit vector */
  471     costh = (b_sp[2] - start[2]) / *ri;
  472     st.site = &emlrtRSI;
  473     b_st.site = &l_emlrtRSI;
  474     sinth = 1.0 - costh * costh;
  475     st.site = &emlrtRSI;
  476     b_sqrt(&st, &sinth);
  477     dp = *ri * sinth;
  478     sinphi = (b_sp[1] - start[1]) / dp;
  479     cosphi = (b_sp[0] - start[0]) / dp;
  480     Eds[0].re = cosphi * costh;
  481     Eds[0].im = 0.0;
  482     Eds[1].re = sinphi * costh;
  483     Eds[1].im = 0.0;
  484     Eds[2].re = -sinth;
  485     Eds[2].im = 0.0;
  486     Fp[0].re = -sinphi;
  487     Fp[0].im = 0.0;
  488     Fp[1].re = cosphi;
  489     Fp[1].im = 0.0;
  490     Fp[2].re = 0.0;
  491     Fp[2].im = 0.0;
  492     ET_re = 0.0;
  493     ET_im = 0.0;
  494     EH_re = 0.0;
  495     EH_im = 0.0;
  496     for (i = 0; i < 3; i++) {
  497       a[i].re = Eds[i].re;
  498       a[i].im = -0.0;
  499       ET_re += a[i].re * Einc[i].re - a[i].im * Einc[i].im;
  500       ET_im += a[i].re * Einc[i].im + a[i].im * Einc[i].re;
  501       a[i].re = Fp[i].re;
  502       a[i].im = -0.0;
  503       EH_re += a[i].re * Einc[i].re - a[i].im * Einc[i].im;
  504       EH_im += a[i].re * Einc[i].im + a[i].im * Einc[i].re;
  505       ss[i] = fin[i] - b_sp[i];
  506     }
  507   
  508     EH_re = -EH_re;
  509     EH_im = -EH_im;
  510     sinphi = norm(ss);
  511     y = 1.0 / sinphi;
  512     for (i1 = 0; i1 < 3; i1++) {
  513       ss[i1] *= y;
  514     }
  515   
  516     /*  diffuse scattering unit vector */
  517     costh = (fin[2] - b_sp[2]) / sinphi;
  518     st.site = &b_emlrtRSI;
  519     b_st.site = &l_emlrtRSI;

  521     st.site = &b_emlrtRSI;

  542     Fp[2].im = 0.0;

  562     y = 0.0;

  577     sinphi = 0.0;
  578     sinth = 0.0;
  579     for (i = 0; i < 3; i++) {

  583       sinth += ss[i] * -si[i];
  584       sr[i] = costh;

  593     }

  597       denom = 3.1415926535897931;
  598       num = cosphi;
  599     }
  600   
  601     /*  if ((bl ~= 0) && (Model(bl) == 2)) */
  602     i1 = Model->size[0];
  603     i = (int32_T)sf[0];
  604     if (!((i >= 1) && (i <= i1))) {
  605       emlrtDynamicBoundsCheckR2012b(i, 1, i1, &g_emlrtBCI, sp);
  606     }
  607   
  608     if (Model->data[i - 1] == 2.0) {
  609       i1 = Ar->size[0];
  610       i = (int32_T)sf[0];
  611       if (!((i >= 1) && (i <= i1))) {
  612         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &h_emlrtBCI, sp);
  613       }
  614   
  615       st.site = &c_emlrtRSI;
  616       denom = Falpha(&st, Ar->data[i - 1], -y);
  617       i1 = Ar->size[0];
  618       i = (int32_T)sf[0];
  619       if (!((i >= 1) && (i <= i1))) {
  620         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &i_emlrtBCI, sp);
  621       }
  622   
  623       st.site = &d_emlrtRSI;
  624       num = lobe(&st, sinphi, Ar->data[i - 1]);
  625   
  626       /*  elseif ((bl == 0) && (GModel == 2)) */
  627       /*      denom = Falpha(GAr,costi); */
  628       /*      num = lobe(cosfr,GAr); */
  629     }

  631     /*  if ((bl ~= 0) && (Model(bl) == 3)) */
  632     i1 = Model->size[0];
  633     i = (int32_T)sf[0];
  634     if (!((i >= 1) && (i <= i1))) {
  635       emlrtDynamicBoundsCheckR2012b(i, 1, i1, &j_emlrtBCI, sp);
  636     }
  637   
  638     if (Model->data[i - 1] == 3.0) {
  639       i1 = Lam->size[0];
  640       i = (int32_T)sf[0];
  641       if (!((i >= 1) && (i <= i1))) {

  644   
  645       i1 = Ar->size[0];
  646       b_sf = (int32_T)sf[0];

  649       }
  650   
  651       i1 = Lam->size[0];
  652       i2 = (int32_T)sf[0];

  663       st.site = &e_emlrtRSI;
  664       denom = Lam->data[i - 1] * Falpha(&st, Ar->data[b_sf - 1], -y) + (1.0 -
  665         Lam->data[i2 - 1]) * Falpha(&st, Aii->data[i3 - 1], -y);
  666       i1 = Lam->size[0];
  667       i = (int32_T)sf[0];
  668       if (!((i >= 1) && (i <= i1))) {
  669         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &o_emlrtBCI, sp);
  670       }
  671   
  672       i1 = Ar->size[0];
  673       b_sf = (int32_T)sf[0];
  674       if (!((b_sf >= 1) && (b_sf <= i1))) {
  675         emlrtDynamicBoundsCheckR2012b(b_sf, 1, i1, &p_emlrtBCI, sp);
  676       }
  677   
  678       i1 = Lam->size[0];
  679       i2 = (int32_T)sf[0];
  680       if (!((i2 >= 1) && (i2 <= i1))) {
  681         emlrtDynamicBoundsCheckR2012b(i2, 1, i1, &q_emlrtBCI, sp);
  682       }
  683   
  684       i1 = Aii->size[0];
  685       i3 = (int32_T)sf[0];
  686       if (!((i3 >= 1) && (i3 <= i1))) {
  687         emlrtDynamicBoundsCheckR2012b(i3, 1, i1, &r_emlrtBCI, sp);
  688       }
  689   
  690       st.site = &f_emlrtRSI;
  691       num = Lam->data[i - 1] * lobe(&st, sinphi, Ar->data[b_sf - 1]) + (1.0 -
  692         Lam->data[i2 - 1]) * Falpha(&st, Aii->data[i3 - 1], sinth);
  693   
  694       /*  elseif ((bl == 0) && (GModel == 3)) */
  695       /*      denom = GLam*Falpha(GAr,costi)+(1-GLam)*Falpha(GAii,costi); */
  696       /*      num = GLam*lobe(cosfr,GAr)+(1-GLam)*Falpha(GAii,cosfi); */
  697     }
  698   
  699     x = -y * num * surf / denom;
  700     st.site = &g_emlrtRSI;
  701     if (x < 0.0) {
  702       b_st.site = &n_emlrtRSI;
  703       b_error(&b_st);
  704     }
  705   
  706     for (i1 = 0; i1 < 3; i1++) {
  707       sr[i1] += b_sp[i1];
  708     }
  709   
  710     i1 = vnfaces->size[0];
  711     i = (int32_T)sf[0];
  712     if (!((i >= 1) && (i <= i1))) {
  713       emlrtDynamicBoundsCheckR2012b(i, 1, i1, &c_emlrtBCI, sp);
  714     }
  715   
  716     /*  Eout      = RefField(-1,Re2,Re3,Re4,Re5,vn_temp,Einc); */
  717     /*  Eout = RefField(-1,bl,start,frp,sp,vn_temp,Einc); */
  718     st.site = &h_emlrtRSI;
  719   
  720     /*  Calculates the reflected field */
  721     /*  blo : block number for which the reflected field is evaluated */
  722     /*  E : transmitter position */
  723     /*  R : receiver position */
  724     /*  qr : reflection point */
  725     /*  n : ext. norm. */
  726     /*  Ein : incident field */
  727     /*  Eout : field at the observation point (R) */
  728     /*  Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at> */
  729     /*  History: 2011.9.26 */
  730     /*  Updated by: Xuhong Li <xuhong@ftw.at> */
  731     /*  History: 2013.08.01 */
  732     /*           2014.02.06(save the reflection coefficients) */
  733     /*  global CEP eta GCEP Geta GRcoeff K Rcoeff vnfaces wedges; */
  734     /* %%%%%%%% Incident wave coordinate system calculation : eti & ehi %%%%%%%%% */
  735     /*  eti is the perpendicular vector to the incident plane */
  736     /*  ehi is the parallel vector to the incident plane */
  737     for (i = 0; i < 3; i++) {
  738       c_sp[i] = b_sp[i] - start[i];
  739     }
  740   
  741     y = 1.0 / norm(c_sp);
  742     for (i = 0; i < 3; i++) {
  743       vr[i] = sr[i] - b_sp[i];
  744       c_sp[i] = sr[i] - b_sp[i];
  745       start[i] = y * (b_sp[i] - start[i]);
  746     }
  747   
  748     dr = norm(c_sp);
  749     y = 1.0 / dr;
  750     for (i1 = 0; i1 < 3; i1++) {
  751       vr[i1] *= y;
  752     }
  753   
  754     si[0] = start[1] * vnfaces->data[(((int32_T)sf[0] + vnfaces->size[0] *
  755       ((int32_T)sf[1] - 1)) + (vnfaces->size[0] * vnfaces->size[1] << 1)) - 1] -
  756       start[2] * vnfaces->data[(((int32_T)sf[0] + vnfaces->size[0] * ((int32_T)sf
  757       [1] - 1)) + vnfaces->size[0] * vnfaces->size[1]) - 1];
  758     si[1] = start[2] * vnfaces->data[((int32_T)sf[0] + vnfaces->size[0] *
  759       ((int32_T)sf[1] - 1)) - 1] - start[0] * vnfaces->data[(((int32_T)sf[0] +
  760       vnfaces->size[0] * ((int32_T)sf[1] - 1)) + (vnfaces->size[0] * vnfaces->
  761       size[1] << 1)) - 1];
  762     si[2] = start[0] * vnfaces->data[(((int32_T)sf[0] + vnfaces->size[0] *
  763       ((int32_T)sf[1] - 1)) + vnfaces->size[0] * vnfaces->size[1]) - 1] - start[1]
  764       * vnfaces->data[((int32_T)sf[0] + vnfaces->size[0] * ((int32_T)sf[1] - 1)) -
  765       1];
  766     sinth = norm(si);
  767     b_st.site = &y_emlrtRSI;
  768     c_st.site = &ab_emlrtRSI;
  769     d_st.site = &l_emlrtRSI;
  770     csfi2.re = sinth * sinth;
  771     fac.re = 1.0 - csfi2.re;
  772     fac.im = 0.0;
  773     c_sqrt(&fac);
  774     denom = muDoubleScalarMax(0.0, fac.re);
  775     if (norm(si) <= 1.0E-6) {
  776       /*  if incident wave is normal to the surface, i.e. (cross-product = 0) <=> (theta_inc = 0) */
  777       b_st.site = &x_emlrtRSI;
  778       c_st.site = &bb_emlrtRSI;
  779       emlrt_synchGlobalsToML(&c_st);
  780       b_y = NULL;
  781       m0 = emlrtCreateCharArray(2, iv0);
  782       emlrtInitCharArrayR2013a(&c_st, 7, m0, &u[0]);
  783       emlrtAssign(&b_y, m0);
  784       c_y = NULL;
  785       m0 = emlrtCreateCharArray(2, iv1);
  786       emlrtInitCharArrayR2013a(&c_st, 24, m0, &formatSpec[0]);
  787       emlrtAssign(&c_y, m0);
  788       d_st.site = &eb_emlrtRSI;
  789       emlrt_marshallIn(&d_st, c_feval(&d_st, b_y, emlrt_marshallOut(1.0), c_y,
  790         &d_emlrtMCI), "feval");
  791       emlrt_synchGlobalsFromML(&c_st);
  792       emlrt_synchGlobalsToML(&st);
  793       b_st.site = &db_emlrtRSI;
  794       keyboard(&b_st, &c_emlrtMCI);
  795       emlrt_synchGlobalsFromML(&st);
  796       dp = muDoubleScalarAtan2(-start[2], muDoubleScalarHypot(-start[0], -start[1]));
  797       sinphi = muDoubleScalarAtan2(-start[1], -start[0]);
  798       dv0[0] = muDoubleScalarCos(1.5707963267948966 - sinphi);
  799       dv0[3] = muDoubleScalarSin(1.5707963267948966 - sinphi);
  800       dv0[6] = 0.0;
  801       dv0[1] = -muDoubleScalarSin(1.5707963267948966 - sinphi);
  802       dv0[4] = muDoubleScalarCos(1.5707963267948966 - sinphi);
  803       dv0[7] = 0.0;
  804       for (i1 = 0; i1 < 3; i1++) {
  805         dv0[2 + 3 * i1] = iv2[i1];
  806         dv1[3 * i1] = iv3[i1];
  807       }
  808   
  809       dv1[1] = 0.0;
  810       dv1[4] = muDoubleScalarCos(3.1415926535897931 - dp);
  811       dv1[7] = muDoubleScalarSin(3.1415926535897931 - dp);
  812       dv1[2] = 0.0;
  813       dv1[5] = -muDoubleScalarSin(3.1415926535897931 - dp);
  814       dv1[8] = muDoubleScalarCos(3.1415926535897931 - dp);
  815   
  816       /*  (OR): unitary parallel vector (transmission field) --- 1st interface */
  817       for (i1 = 0; i1 < 3; i1++) {
  818         for (i = 0; i < 3; i++) {
  819           dv2[i1 + 3 * i] = 0.0;
  820           for (b_sf = 0; b_sf < 3; b_sf++) {
  821             dv2[i1 + 3 * i] += dv0[i1 + 3 * b_sf] * dv1[b_sf + 3 * i];
  822           }
  823         }
  824   
  825         for (i = 0; i < 3; i++) {
  826           e[i1 + 3 * i] = 0.0;
  827           for (b_sf = 0; b_sf < 3; b_sf++) {
  828             e[i1 + 3 * i] += dv2[i1 + 3 * b_sf] * (real_T)b[b_sf + 3 * i];
  829           }
  830         }
  831   
  832         si[i1] = e[i1];
  833         fin[i1] = e[6 + i1];
  834       }
  835   
  836       /*  (OR): unitary perpendicular vector (transmission field) --- 1st interface */
  837       /*  (OR): eti -> unitary parallel vector */
  838       /*        ehi -> unitary perpendicular vector */
  839       /*  %     eti    = [0 0 1].'; */
  840       /*       */
  841       /*      d          = zeros(1,6); */
  842       /*      for i = 1:6 */
  843       /*          d(1,i) = sum(n_re-squeeze(vnfaces(blo,i,:))); */
  844       /*      end */
  845       /*      [~,ind]     = min(abs(d)); */
  846       /*       */
  847       /*      P1          = reshape(wedges(blo,ind,1,:),3,1); */
  848       /*      P2          = reshape(wedges(blo,ind,2,:),3,1); */
  849       /*      P4          = reshape(wedges(blo,ind,4,:),3,1); */
  850       /*       */
  851       /*      if abs(P1(1)-P2(1)) > abs(P1(1)-P4(1)) */
  852       /*          P       = [P1 P2]; */
  853       /*      else */
  854       /*          P       = [P1 P4]; */
  855       /*      end */
  856       /*      [~,ind]     = max(P(1,:)); */
  857       /*       */
  858       /*      if (acosd(dot(vi,n)) < 90) % incident wave and normal vector to the surface have same direction */
  859       /*          eti     = P(:,ind)-P(:,setdiff(1:2,ind)); */
  860       /*      else % incident wave and normal vector to the surface have opposite direction */
  861       /*          eti     = P(:,setdiff(1:2,ind))-P(:,ind); */
  862       /*      end */
  863     } else {
  864       y = 1.0 / norm(si);
  865       for (i1 = 0; i1 < 3; i1++) {
  866         si[i1] *= y;
  867       }
  868   
  869       fin[0] = si[1] * start[2] - si[2] * start[1];
  870       fin[1] = si[2] * start[0] - si[0] * start[2];
  871       fin[2] = si[0] * start[1] - si[1] * start[0];
  872     }
  873   
  874     /* %%%%%%%%%%%%%%%% E components in the SW-fixed coordinate system %%%%%%%%%%%%%%% */
  875     i1 = CEP->size[0];
  876     i = (int32_T)sf[0];
  877     if (!((i >= 1) && (i <= i1))) {
  878       emlrtDynamicBoundsCheckR2012b(i, 1, i1, &s_emlrtBCI, &st);
  879     }
  880   
  881     if (CEP->data[i - 1] == 1.0) {
  882       costh = -1.0;
  883       num = 0.0;
  884   
  885       /*  -> perpendicular polarization coefficient */
  886       csfi2.re = 1.0;
  887       csfi2.im = 0.0;
  888   
  889       /*  -> parallel polarization coefficient */
  890     } else {
  891       i1 = eta->size[0];
  892       i = (int32_T)sf[0];
  893       if (!((i >= 1) && (i <= i1))) {
  894         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &t_emlrtBCI, &st);
  895       }
  896   
  897       csfi2.re = eta->data[i - 1].re - csfi2.re;
  898       i1 = eta->size[0];
  899       i = (int32_T)sf[0];
  900       if (!((i >= 1) && (i <= i1))) {
  901         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &t_emlrtBCI, &st);
  902       }
  903   
  904       csfi2.im = eta->data[i - 1].im;
  905       i1 = eta->size[0];
  906       i = (int32_T)sf[0];
  907       if (!((i >= 1) && (i <= i1))) {
  908         emlrtDynamicBoundsCheckR2012b(i, 1, i1, &emlrtBCI, &st);
  909       }
  910   
  911       dp = eta->data[(int32_T)sf[0] - 1].re * eta->data[(int32_T)sf[0] - 1].re -
  912         eta->data[(int32_T)sf[0] - 1].im * eta->data[(int32_T)sf[0] - 1].im;
  913       sinphi = eta->data[(int32_T)sf[0] - 1].re * eta->data[(int32_T)sf[0] - 1].im
  914         + eta->data[(int32_T)sf[0] - 1].im * eta->data[(int32_T)sf[0] - 1].re;
  915       if (sinphi == 0.0) {
  916         if (csfi2.im == 0.0) {
  917           Ch.re = csfi2.re / dp;
  918           Ch.im = 0.0;
  919         } else if (csfi2.re == 0.0) {
  920           Ch.re = 0.0;
  921           Ch.im = csfi2.im / dp;
  922         } else {
  923           Ch.re = csfi2.re / dp;
  924           Ch.im = csfi2.im / dp;
  925         }
  926       } else if (dp == 0.0) {
  927         if (csfi2.re == 0.0) {
  928           Ch.re = csfi2.im / sinphi;
  929           Ch.im = 0.0;
  930         } else if (csfi2.im == 0.0) {
  931           Ch.re = 0.0;
  932           Ch.im = -(csfi2.re / sinphi);
  933         } else {
  934           Ch.re = csfi2.im / sinphi;
  935           Ch.im = -(csfi2.re / sinphi);
  936         }
  937       } else {
  938         sinth = muDoubleScalarAbs(dp);
  939         cosphi = muDoubleScalarAbs(sinphi);
  940         if (sinth > cosphi) {
  941           cosphi = sinphi / dp;
  942           dp += cosphi * sinphi;
  943           Ch.re = (csfi2.re + cosphi * csfi2.im) / dp;
  944           Ch.im = (csfi2.im - cosphi * csfi2.re) / dp;
  945         } else if (cosphi == sinth) {
  946           if (dp > 0.0) {
  947             cosphi = 0.5;
  948           } else {
  949             cosphi = -0.5;
  950           }
  951   
  952           if (sinphi > 0.0) {
  953             dp = 0.5;
  954           } else {
  955             dp = -0.5;
  956           }
  957   
  958           Ch.re = (csfi2.re * cosphi + csfi2.im * dp) / sinth;
  959           Ch.im = (csfi2.im * cosphi - csfi2.re * dp) / sinth;
  960         } else {
  961           cosphi = dp / sinphi;
  962           dp = sinphi + cosphi * dp;
  963           Ch.re = (cosphi * csfi2.re + csfi2.im) / dp;
  964           Ch.im = (cosphi * csfi2.im - csfi2.re) / dp;
  965         }
  966       }
  967   
  968       fac = csfi2;
  969       c_sqrt(&fac);
  970       c_sqrt(&csfi2);
