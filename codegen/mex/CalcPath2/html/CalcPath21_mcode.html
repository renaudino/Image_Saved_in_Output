1function [FSLN, PSLN, Reflex, RF, RP, NbrDSSlns, NbrSlns] = CalcPath2 (tx, rx, ordmax, rvsds, FSLN, ...
  2                                                                           PSLN, RF, RP, NbrDSSlns, NbrSlns)
  3% function CalcPath(tx,rx,ordmax,rvsds)
  4
  5% #codegen
  6% Calculates the path followed by a ray, i.e. the reflection points
  7% 
  8% Authors: Francesco Mani <francesco.mani@uclouvain.be>
  9%          Mingming Gan   <gan@ftw.at>
 10% 
 11% History: 2011.09.26
 12%          2011.10.04 -> updated after learning the diffuse scattering
 13
 14global vnfaces wedges trun2;
 15
 16% Nblock  = p.Nblock;
 17% vnfaces = p.vnfaces;
 18% wedges  = p.wedges;
 19% 
 20% MSKFlag = p.MSKFlag;
 21% CEP     = p.CEP;
 22
 23ornum = 1;
 24pass  = 1;
 25count = 0;
 26
 27fface = zeros(2,1);
 28ind   = find(trun2(:,7) == ordmax);
 29curs  = trun2(ind(ornum),:);
 30right = curs;
 31
 32persistent oldReflex;
 33
 34if isempty(oldReflex)
 35    oldReflex = 0;
 36end
 37Reflex     = 0;
 38
 39% Atemp      = zeros(3);
 40
 41track      = zeros(10,3);
 42trackface  = zeros(10,2);
 43
 44vnfaces_re = zeros(3,1);
 45pointV     = zeros(3,1);
 46
 47while (isempty(right) ~= 1)
 48    back = right;
 49    extrem = rx;
 50    glob = 1;
 51    m = 1;
 52    
 53    while (back(1) ~= 0)
 54        bt1 = reshape(back(2:3),1,2);
 55        bt2 = reshape(back(4:6),3,1);
 56        bt3 = reshape(extrem,3,1);
 57        pointV(:,1) = Breakthrough(bt1,bt2,bt3);
 58%         pointV         = Breakthrough(back(2:3).',back(4:6).',extrem);
 59        track(m,:)     = pointV; % coordinates of reflection point
 60        trackface(m,:) = back(2:3);
 61        
 62        difftest = extrem-pointV;
 63%         if (back(2) ~= 0)
 64            vnfaces_re(:) = vnfaces(trackface(m,1),trackface(m,2),:);
 65            if ((vnfaces_re'*difftest) <= 0) %(dot(vnfaces_re,difftest) <= 0)
 66                glob = 0;
 67                break;
 68            end
 69%         end
 70        
 71%         if (back(2) == 0)
 72%             app      = AppGround(pointV,p);
 73%         else
 74            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 75%             Atemp(:) = wedges(back(2),back(3),[1 2 4],:);
 76%             app      = AppFace(pointV,Atemp);
 77            app      = AppFace2(pointV,squeeze(wedges(back(2),back(3),:,:)).');
 78            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 79%         end
 80        
 81        if (app == 1)
 82            cache = Blocking(extrem,pointV);
 83            if (cache == 1)
 84                glob = 0; break;
 85            end
 86            
 87            fface(1) = back(2);
 88            if (fface(1) > 0)
 89                for j = 1:6
 90                    if (j ~= back(3))
 91                        fface(2) = j;
 92                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 93%                         Atemp(:) = wedges(fface(1),fface(2),[1 2 4],:);
 94%                         appf = AppFace(pointV,Atemp);
 95                        appf = AppFace2(pointV,squeeze(wedges(fface(1),fface(2),:,:)).');
 96                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 97                        if (appf == 1)
 98                            glob = 0; break;
 99                        end
100                    end
101                end
102                
103                if (sum(abs(extrem-pointV) < 1e-12) == length(extrem)) %(abs(extrem-pointV) < 1.0e-12)
104                    glob = 0; break;
105                end
106            end
107        else
108            glob = 0;
109            break;
110        end
111        extrem(:,1) = pointV;
112        back = trun2(back(1),:);
113        m = m+1; % means that the path found is okay, then go to the next level
114    end
115    
116    if (glob == 1)
117        cache = Blocking(tx(:),pointV(:));
118        if (cache == 1)
119            glob = 0;
120        end
121    end
122    if (glob == 1)
123        Reflex = 1;
124%         if (rvsds == 0) % (OR): commented -> 28.04.2017
125        if ~all(rvsds)
126            NbrSlns(ordmax) = pass;
127            for i = 1:m-1
128                PSLN(ordmax,pass,i,:) = track(m-i,:);
129                FSLN(ordmax,pass,i,:) = trackface(m-i,:);
130            end
131            pass = pass+1;
132            count = count+1;
133        else
134            NbrDSSlns(ordmax+1) = pass;
135            for i = 1:m-1
136                RF(ordmax+1,pass,i,:) = trackface(m-i,:);
137                RP(ordmax+1,pass,i,:) = track(m-i,:);
138            end
139            pass  = pass+1;
140            count = count+1;
141        end
142    end
143    ornum = ornum+1;
144    if (ornum <= length(ind))
145        right = trun2(ind(ornum),:);
146    else
147        break;
148    end
149end
150if (Reflex == 0)
151%     if (rvsds == 0) % (OR): commented -> 28.04.2017
152    if ~all(rvsds)
153        NbrSlns(ordmax) = 0;
154    else
155        NbrDSSlns(ordmax) = 0;
156    end
157end
158if (count ~= 0) || (oldReflex == 1) || (Reflex == 1)
159    Reflex = 1;
160else
161%     if (rvsds == 0) % (OR): commented -> 28.04.2017
162    if ~all(rvsds)
163        NbrSlns(ordmax) = 0;
164    else
165        NbrDSSlns(ordmax) = 0;
166    end
167end
168oldReflex = Reflex;
