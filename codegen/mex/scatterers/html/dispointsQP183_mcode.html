1function [points, number, space] = dispointsQP (b, f, op)
  2
  3% Generates quasi randomly distributed scattering tiles without overlapping on each surface
  4% 
  5% Authors: Xuhong Li <xuhong@ftw.at>
  6% 
  7% History: 2013.10.01
  8
  9global wedges vnfaces lambda;
 10
 11point_temp = zeros(3,1);
 12area_temp  = 0;
 13
 14% Based on far-field condition, decide a proper tile size
 15for aa = 1:4
 16    point_temp(1:3) = wedges(b,f,aa,:);
 17    d               = norm(op-point_temp);
 18    edge            = sqrt(d*lambda/2);
 19    area_temp       = area_temp+edge;
 20end
 21area = pi*(area_temp/8)^2;
 22radi = 2*sqrt(area/pi);   % assume the area is circular
 23
 24number = 0;               % number of scatterers on this surface
 25
 26normal      = zeros(3,1);
 27normal(:,1) = vnfaces(b,f,:);
 28
 29points = zeros(10000,3);
 30space  = 0;
 31
 32p1 = reshape(wedges(b,f,1,:),3,1);
 33p2 = reshape(wedges(b,f,2,:),3,1);
 34p3 = reshape(wedges(b,f,3,:),3,1);
 35p4 = reshape(wedges(b,f,4,:),3,1);
 36
 37spacelimit = [min([p1 p2 p3 p4],[],2) max([p1 p2 p3 p4],[],2)];
 38
 39% center = op;
 40% a = 1;
 41% while (center == op)
 42%     fprintf('   [%d %d %d] \n',b,f,a);
 43    center     = p1+(p2-p1)*rand(1)+(p4-p1)*rand(1);
 44%     a = a+1; [op center]
 45% end
 46
 47d1 = norm(center-p1);
 48d2 = norm(center-p2);
 49d3 = norm(center-p3);
 50d4 = norm(center-p4);
 51
 52dis_max = max([d1 d2 d3 d4]);
 53n       = floor(dis_max/radi); % number of number of concentric circles
 54
 55if (n == 0) || (n == 1)
 56    points    = zeros(1,3);
 57    points(:) = (wedges(b,f,1,:)+wedges(b,f,2,:)+wedges(b,f,3,:)+wedges(b,f,4,:))/4;
 58    number    = 1;
 59    
 60elseif (n > 1)
 61    number_sub = 0;
 62    i = 1;
 63    
 64    while (i <= n) % loop for each circle
 65        theta = rand(1)*2*pi;     % random start angle
 66        angle = 2*asin(1/(2*i));  % angle resolution
 67        np    = ceil(2*pi/angle);
 68        count = 0;
 69        
 70        for m = 0:np-1
 71            a    = theta+m*angle;
 72            k    = normal/norm(normal);
 73            v    = center+(p1-center)/norm(p1-center)*radi*i;
 74            vrot = center+(v-center)*cos(a)+cross(k,(v-center))*sin(a); % Rodrigues' rotation formula
 75            
 76            if (vrot(1) < spacelimit(1,2) && vrot(1) > spacelimit(1,1) && ...
 77                    vrot(2) < spacelimit(2,2) && vrot(2) > spacelimit(2,1) && ...
 78                    vrot(3) < spacelimit(3,2) && vrot(3) > spacelimit(3,1))
 79                number_sub           = number_sub+1;
 80                points(number_sub,:) = vrot.';
 81                count                = count+1;
 82            end
 83        end
 84        i = i+1;
 85    end
 86    number_sub           = number_sub+1;
 87    
 88    points(number_sub,:) = center;       % add 'center' as a scatterer
 89    number               = number_sub;
 90end
 91temp = number*area/norm(cross(p3-p1,p4-p2))*2;%(sqrt(line(1)^2+line(2)^2)*line(3));%(line(1)*line(2)); % compare the sum of tile size with the surface area
 92
 93if (temp <= 1) %  if the ratio 'temp <= 1', then use default area size
 94    space = ones(number,1)*area;
 95    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 96    %%%%%%  force the area ratio = 1  %%%%%%%
 97    %     Npe_area = Npe_area+1;            %
 98    %     pe_area(Npe_area) = 1;            %
 99    %     Area = line(1)*line(2)/number;    %
100    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
101elseif (temp > 1) %  otherwise, use averaged area size
102    space = ones(number,1)*(norm(cross(p3-p1,p4-p2))/2/number);%(line(1)*line(2)/number);
103end
104
105points(number+1:end,:) = [];
