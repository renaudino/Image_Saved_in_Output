1function Construct(OrdRef)
 2% Build the image tree
 3
 4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 5% History: 2011.9.21
 6
 7% Updated by: Xuhong Li <xuhong@ftw.at>
 8% History: 2013.08.01
 9% Build the image tree in level order from 1 to OrdRef+1, instead of using recursion 
10% Mtrun: 1   -----   flag, empty or not
11%        2~3 -----   face, 2--block index 3--surface index
12%        4~6 -----   point coordinates
13%        7   -----   Nbfaces, number of images of current point 
14%        8~9 -----   previous level
15
16global Nbrimg Mtrun;
17mid = zeros(2,1);
18M = zeros(OrdRef+1,1); % image points number in each level
19M(1) = 1;              
20for i = 1:OrdRef
21   a=0;
22   b=0;
23   mid(1) = i;       % level
24   for j = 1:M(i)
25       mid(2) = j;   % point index
26       pnbr = PointsImages(mid); % Number of image points for current point
27       Nbrimg(i+1) = Nbrimg(i+1)+pnbr;
28       Mtrun(i,j,7) = pnbr;    
29       
30       if j == 1      % corresponding images points index in next level
31               a=1;
32               b=pnbr;
33       else
34               a=b+1;
35               b=b+pnbr;
36       end
37       
38       Mtrun(i+1,a:b,8) = i;   %  previous level
39       Mtrun(i+1,a:b,9) = j;   %  previous level
40       M(i+1) = M(i+1)+pnbr;
41       
42   end      
43end
