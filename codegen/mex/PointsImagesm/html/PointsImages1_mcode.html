1function pnbr = PointsImages(mid)
  2% In order to find the coordinates of the image points in the same order
  3
  4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
  5% History:  2011.9.21
  6
  7% Updated by: Xuhong Li <xuhong@ftw.at>
  8% History: 2013.08.01
  9
 10global wedges Nblock DBIMAGES vnfaces Nbrimg Mtrun Nreflx recept;
 11
 12m = (mid(1)+1);
 13n = Nbrimg(m);
 14v = zeros(3,1);
 15no = zeros(3,1);
 16cnt = zeros(3,1);
 17E = zeros(3,1);
 18pointV = zeros(3,1);
 19wedges_re = zeros(3,1);
 20vnfaces_re = zeros(3,1);
 21p = zeros(3,1);
 22
 23
 24E(:,1) = Mtrun(mid(1),mid(2),4:6);  % point
 25face = Mtrun(mid(1),mid(2),2:3);    % face
 26
 27
 28if (face(1) == -1)   %   if this is Tx
 29        cnt(:,1) = E;
 30else
 31    if(face(1) ~= 0)  %  if this is not ground
 32        cnt(:,1) = sum(wedges(face(1),face(2),:,:),3)/4;   
 33        % the coordinates of the center point of the surface realted to the image
 34    end
 35end
 36
 37for i = 1:Nblock
 38    if i ~= face(1)    % make sure that the next reflection is not in the same surface as the previous one
 39        for j = 1:6
 40            for l = 1:4
 41                wedges_re(:,1) = wedges(i,j,l,:);
 42                if (face(1) ~= 0)
 43                    v = (v + (wedges_re - cnt));
 44                else
 45                    v = (v + (wedges_re - E));
 46                end 
 47            end
 48            
 49            v = v/4; 
 50            
 51            if(face(1) == 0)    % the privoius relection from ground
 52                resu1 = -1;
 53                
 54            else
 55                vnfaces_re(:,1) = vnfaces(i,j,:);
 56                resu1 = v'*vnfaces_re;
 57                
 58            end
 59            if (resu1 < 0)
 60                n = n + 1;
 61
 62                Mtrun(m,n,:)=DBIMAGES;
 63                Mtrun(m,n,2:3)=[i,j];
 64
 65                no(:,1) = vnfaces(i,j,:);
 66                p(:,1) = wedges(i,j,1,:);
 67                pt = p - E;
 68                Mtrun(m,n,4:6)=E + 2*(no'*pt).*no;
 69                
 70% -------------- remove useless points on the last level ---------------- %                
 71                if m == Nreflx+1   % the highest order of reflection
 72                    
 73                    pointV(:,1) = Breakthrough(Mtrun(m,n,2:3),Mtrun(m,n,4:6),recept);    % the coordinates of the reflection point       
 74                    if (Mtrun(m,n,2) == 0)
 75                        app = AppGround(pointV);   % check the relection point belongs to ground or not
 76                    else
 77                        app = AppFace(pointV,Mtrun(m,n,2:3));    % check the relection point belongs to surface or not
 78                    end  
 79                    
 80                    if (app ~= 1)    % the point does not belongs to ground/surface
 81                        n = n-1;
 82                    end
 83                end
 84% ----------------------------------------------------------------------- %                
 85            end
 86        end
 87    end
 88end 
 89
 90no(1) = 0;    % ground case
 91no(2) = 0;
 92no(3) = 1;
 93
 94pt = -E;
 95
 96if (face(1) ~= 0) && (cnt(3)> 0)
 97    n = n + 1;
 98
 99    Mtrun(m,n,:)=DBIMAGES;
100    Mtrun(m,n,2:3)=[0;0];
101    Mtrun(m,n,4:6)=E + 2*(no'*pt).*no;
102
103end 
104pnbr = n-Nbrimg(m);
105    
106    
107                    
108
