1function appground = AppGround(gpoint)
 2% Calculates if the point belongs to the ground
 3
 4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 5% History:  2011.9.19
 6
 7global Nblock;
 8face1 = zeros(2,1);
 9face1(2) = 5;
10
11appground = 1;
12for i = 1 : Nblock
13    face1(1) = i;
14    appface = AppFace(gpoint,face1);
15    if (appface == 1)
16        appground = 0;
17        break; 
18    end 
19end
