1function flag = BlockCheck(bl,p_re)
 2% Checks if the penetration point belongs to some other block
 3
 4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 5% History: 2011.9.27
 6
 7% Updated by: Xuhong Li <xuhong@ftw.at>
 8% History: 2013.08.01
 9
10global Nblock wedges;
11
12p=zeros(3,1);
13res=0;
14
15p(:,1) = p_re;
16flag = -1;
17f = zeros(2,1);
18for i = 1 : Nblock
19    if (i ~= bl)
20        for j = 1:6
21            f(1) = i;
22            f(2) = j;
23            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
24%             res = AppFace(p,f);
25            res = AppFace2(p,reshape(wedges(f(1),f(2),:,:),[],3).');
26            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
27            if (res == 1)
28                flag = i;
29            end 
30        end 
31    end 
32end
