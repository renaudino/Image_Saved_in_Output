1function [et, eh] = cart2rayfixed (s)
 2
 3[az,el,~]           = cart2sph(-s(1),-s(2),-s(3));
 4
 5Raz                 = [cos(pi/2-az) sin(pi/2-az) 0 ; -sin(pi/2-az) cos(pi/2-az) 0 ; 0 0 1];
 6Rel                 = [1 0 0 ; 0  cos(pi-el) sin(pi-el) ; 0 -sin(pi-el) cos(pi-el)];
 7
 8e                  = Raz*Rel*eye(3);
 9
10et                 = e(:,1); % (OR): unitary parallel vector (transmission field) --- 1st interface
11eh                 = e(:,3); % (OR): unitary perpendicular vector (transmission field) --- 1st interface
