1function EffConstruct(OrdRef,tx,rvsds)
  2% Data construction
  3
  4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
  5% History: 2011.9.21
  6
  7% Updated by: Xuhong Li <xuhong@ftw.at>
  8% History: 2013.08.01
  9% Combine Construct and CalcPath, calculate valid reflection path without
 10% build whole image tree and using recursion: EffConstruct, EffPointsImages
 11% Mpath: 1~2 ----- face,  3~5 ----- point
 12      
 13
 14global vnfaces recept Nblock Reflex NbrSlns NbrDSSlns oldReflex PSLN FSLN RP RF Mpath;
 15
 16x = 0;
 17app = 0;
 18cache = 0;
 19tempvalue = 0;
 20count = 0;
 21oldReflex = Reflex;
 22
 23%--------------- unify variable size for code convertion -----------------%
 24% pass = ones(OrdRef,1);
 25pass = zeros(OrdRef,1);
 26leveltemp = zeros(1,Nblock*6,5);
 27pointVcheck = zeros(3,1);
 28extremcheck = zeros(3,1);
 29pointcheck = zeros(3,1);
 30facecheck = zeros(2,1);
 31Temp3 = zeros(3,1);
 32
 33track = zeros(OrdRef+1,3);
 34trackface = zeros(OrdRef+1,2);
 35vnfaces_re = zeros(3,1);
 36
 37pointV = zeros(3,1);
 38extrem = zeros(3,1);
 39back = zeros(1,3);
 40fface = zeros(2,1);
 41numberuse = zeros(1,Nblock*6);
 42loopdomain = zeros(OrdRef+1,1);
 43%-------------------------------------------------------------------------%
 44
 45loopdomain(1) = 1;  % only Tx in the first level
 46
 47for i = 1:OrdRef   % build up the original image tree
 48    loopdomain(i+1) = EffPointsImages(i);  
 49end
 50
 51extrem(:,1) = recept;
 52b = 1;  % flag
 53golevel = 2;
 54goflag = 0; % original image tree: 0    updated image tree: 1
 55jumplevel = OrdRef+1; % pointer. from which point to update the image tree
 56order = 1;
 57
 58while (jumplevel > 1 && order == 1) % update the image tree until go back to Tx
 59%% part 1
 60% 1. Update the image tree until there are useful imagepoints in the last
 61%    level, then go to part 2
 62% 2. Calculate valid lower order (smaller than 'OrdRef') reflection path after updating image tree
 63
 64if OrdRef == 1  % if reflection order is 1, set order to 0, in order to jump out the loop
 65    order = 0;    
 66end
 67
 68while ( b ==1 )
 69    p = 0;
 70    while (golevel ~= OrdRef+1 && golevel > 1)
 71        
 72        if goflag ==1  
 73        
 74            leveltemp = Mpath(golevel,2:end,:);  % remove the first used point, renew the tree from the second point on this level
 75            Mpath(golevel,:,:) = 0;
 76            Mpath(golevel,1:(Nblock*6-1),:) = leveltemp;
 77        
 78            Mpath(golevel+1:end,:,:) = 0;        % clear the last few levels for new imagepoints
 79            loopdomain(golevel+1:end) = 0;
 80            
 81            for m = golevel+1:OrdRef+1
 82                loopdomain(m) = EffPointsImages(m-1);
 83            end
 84            
 85            loopdomain(golevel) = loopdomain(golevel)-1;
 86        
 87        end
 88     
 89
 90        for n = golevel:OrdRef   % backtrack from the new point
 91            glob = 1;
 92            j = n;
 93            back = Mpath(n,1,:);
 94            extrem(:,1) = recept;
 95            m = 1;
 96            
 97            while (j ~=1 && glob == 1)   
 98                bt1 = reshape(back(1:2),1,2);    %  face
 99                bt2 = reshape(back(3:5),3,1);    % point
100                bt3 = reshape(extrem,3,1);
101                pointV(:,1) = Breakthrough(bt1,bt2,bt3);
102                
103                track(m,:) = pointV;    % coordinates of reflection point
104                trackface(m,:) = back(1:2);    % face
105                difftest = extrem - pointV;
106                
107                if (back(1) ~= 0)
108                    vnfaces_re(:,1) = vnfaces(trackface(m,1),trackface(m,2),:);
109                    
110                    if ((vnfaces_re'*difftest) <= 0)
111                        glob = 0;
112                        break;
113                    end
114                end
115                
116                if (back(1) == 0)
117                    app = AppGround(pointV);
118                else
119                    back_temp=back(1,1:2);
120                    back_temp=back_temp';
121                    app = AppFace(pointV,back_temp); % Attention AppFace input dimension match
122                end
123                
124                if (app == 1)
125                    cache = Blocking(extrem,pointV);
126                    if (cache == 1)
127                        glob = 0;
128                        break;
129                    end
130                    
131                    fface(1) = back(1);   % index of the current block
132                    
133                    if (fface(1) > 0)     
134                        
135                        for Nface = 1:6
136                            if (Nface ~= back(2))
137                                fface(2) = Nface;
138                                appf = AppFace(pointV,fface);
139                                
140                                if (appf == 1)
141                                    glob = 0;
142                                    break;
143                                end
144                            end
145                        end
146                        tempvalue = norm(extrem - pointV);  % test
147                        if (tempvalue < 1.0e-12)
148                            
149                            glob = 0;
150                            break;
151                        end
152                    end
153                    
154                else
155                    glob = 0;
156                    break;
157                end
158                
159                extrem(:,1) = pointV;
160                j = j-1;
161                back = Mpath(j,1,:); % the first point of upper level
162                m = m+1;    %  the reflection point is okay, go to the upper level
163            
164            end    % while (j ~=0)
165            
166            
167            if (glob == 1)
168                Temp3(:,1)=tx;
169                cache = Blocking(Temp3,pointV);   %  input of Blocking should be 3*1, so use tx'
170                
171                if (cache == 1)
172                    glob = 0;
173                end
174            end
175            if (glob == 1)
176                Reflex = 1;
177                if (rvsds == 0)
178                    pass(n-1) = pass(n-1) + 1; % test
179                    NbrSlns(n-1) = pass(n-1);
180                    for i = 1:(m-1)
181                        PSLN(n-1,pass(n-1),i,:) = track(m-i,:);
182                        FSLN(n-1,pass(n-1),i,:) = trackface(m-i,:);
183                    end
184%                     pass(n-1) = pass(n-1) + 1;
185                    count = count + 1;
186                else
187                    pass(n-1) = pass(n-1) + 1; % test
188                    NbrDSSlns(n) = pass(n-1);
189                    
190                    for i = 1:(m-1)
191                        
192                        RP(n,pass(n-1),i,:) = track(m-i,:);
193                        RF(n,pass(n-1),i,:) = trackface(m-i,:);
194                    end
195%                     pass(n-1) = pass(n-1) + 1;
196                    count = count + 1;
197                end
198            end
199            
200            
201        end
202        
203          
204        if goflag == 0
205            
206            golevel = OrdRef+1; 
207            goflag = 1;
208            
209        elseif goflag == 1
210            
211            if loopdomain(OrdRef) == 1 % special case: only one point left on the last second level
212            p = 1;  % flag
213            jumplevel = OrdRef-1;
214            pp = 0;
215            
216            while jumplevel >= 1 && pp == 0 %  backtrack until there is more than one valid point on that level, 
217                
218                if ( loopdomain(jumplevel)==1 )  % important for the next start point
219                    jumplevel = jumplevel-1;
220                    pp = 0;
221                else
222                    pp = 1;
223                end
224            end
225            
226            golevel = OrdRef+1;
227            
228            else
229                
230            p = 0;
231            golevel = OrdRef+1;
232            end
233        
234        end
235    end
236    
237
238    
239    %------check if the point on the last level is useful------% 
240    if golevel == OrdRef+1   
241        
242        x = 0;
243       for iii =1:loopdomain(golevel)
244           facecheck(:,1) = Mpath(golevel,iii,1:2);    % 1*2
245           pointcheck(:,1) = Mpath(golevel,iii,3:5);   % 3*1
246           extremcheck(:,1) = reshape(recept,3,1); %3*1
247           pointVcheck(:,1) = Breakthrough(facecheck',pointcheck,extremcheck);
248           
249           if (facecheck(1) == 0)   % if ground
250               app = AppGround(pointVcheck);
251           else
252               app = AppFace(pointVcheck,facecheck); % Attention AppFace input dimension match
253           end 
254           
255           if app == 1
256               x = x+1;
257               numberuse(x) = iii;   %  the useful imagepoints index on the last level
258           end
259           
260       end
261       
262       if x == 0   %  if no useful imagepoints
263           
264           if p == 1  % update the image tree from the 1st point on jumplevel
265
266               golevel = jumplevel;
267               
268           elseif p == 0
269
270               golevel = golevel-1;
271               
272           end
273           
274           b = 1; % there is no valid point on the last level, continue part 1
275           
276       elseif x >= 1 && x <loopdomain(golevel)    %   renew the OrdRef+1 level
277           
278           for u = 1:x
279               leveltemp(1,u,:) = Mpath(golevel,numberuse(u),:);
280           end
281           
282           Mpath(golevel,:,:) = 0;
283           Mpath(golevel,1:x,:) = leveltemp(1,1:x,:);
284           
285           if p == 1  % update the image tree from the 1st point on jumplevel
286               
287               golevel = jumplevel;
288               
289           elseif p == 0
290
291               golevel = golevel-1;
292               
293           end
294           
295           b = 0; % there are valid points on the last level, go to part 2
296           
297       elseif x == loopdomain(golevel)  %  all imagepoints can be used
298           
299           if p == 1  % update the image tree from the 1st point on jumplevel
300               
301               golevel = jumplevel;
302               
303           elseif p == 0
304
305               golevel = golevel-1;
306               
307           end
308           b = 0;  % there are valid points on the last level, go to part 2
309           
310       
311       end
312
313    end
314  
315    
316end 
317
318loopdomain(OrdRef+1) = x; %  # valid points on the last level 
319
320
321
322
323
324
325
326%% part 2 
327% Backtrack from the points on the last level, calculate full order ('OrdRef') reflection path 
328
329
330    n = 1;
331    while (n <= loopdomain(OrdRef+1) )   % loop for useful image points in the last level
332        glob = 1;
333        j = OrdRef+1;
334        back = Mpath(OrdRef+1,n,:); 
335        m = 1;
336        extrem(:,1) = recept;
337        
338    while (j ~=1 && glob == 1)
339       bt1 = reshape(back(1:2),1,2);    %  face
340       bt2 = reshape(back(3:5),3,1);    % point
341       bt3 = reshape(extrem,3,1);
342       pointV(:,1) = Breakthrough(bt1,bt2,bt3);
343
344       
345        track(m,:) = pointV;    % coordinates of reflection point
346        trackface(m,:) = back(1:2);     % face
347        difftest = extrem - pointV;
348
349        if (back(1) ~= 0)
350            vnfaces_re(:,1) = vnfaces(trackface(m,1),trackface(m,2),:);
351
352            if ((vnfaces_re'*difftest) <= 0)
353                glob = 0;
354                break;
355            end
356        end
357        if (back(1) == 0)
358
359            app = AppGround(pointV);
360        else
361            back_temp=back(1,1:2);
362            back_temp=back_temp';
363            app = AppFace(pointV,back_temp); % Attention AppFace input dimension match
364
365        end 
366    
367        if (app == 1)
368            cache = Blocking(extrem,pointV);
369            if (cache == 1)
370                glob = 0;
371                break;
372            end 
373            
374            fface(1) = back(1);   % index of the current block
375            if (fface(1) > 0)          
376                for Nface = 1:6
377                    if (Nface ~= back(2))
378                        fface(2) = Nface;
379                        appf = AppFace(pointV,fface);
380                        if (appf == 1)
381                            glob = 0;
382                            break;
383                        end
384                    end 
385                end
386                
387                tempvalue = norm(extrem - pointV); 
388                if (tempvalue < 1.0e-12)
389                    glob = 0;
390                    break;
391                end 
392            end 
393        else 
394            glob = 0;
395            break;
396        end
397    
398        extrem(:,1) = pointV;
399        j = j-1;
400        back = Mpath(j,1,:);
401        m = m+1;    
402        
403    end    
404    
405    if (glob == 1)
406        
407        Temp3(:,1)=tx;
408        cache = Blocking(Temp3,pointV);   
409        
410        if (cache == 1)
411            glob = 0;
412        end 
413    end
414    if (glob == 1)
415        Reflex = 1;
416        if (rvsds == 0)
417            pass(OrdRef) = pass(OrdRef) + 1; % test
418            NbrSlns(OrdRef) = pass(OrdRef);
419            for i = 1:(m-1)
420                PSLN(OrdRef,pass(OrdRef),i,:) = track(m-i,:);
421                FSLN(OrdRef,pass(OrdRef),i,:) = trackface(m-i,:);
422            end
423%             pass(OrdRef) = pass(OrdRef) + 1;
424            count = count + 1;
425        else 
426            pass(OrdRef) = pass(OrdRef) + 1; %test
427            NbrDSSlns(OrdRef+1+1) = pass(OrdRef);
428            for i = 1:(m-1)
429                RP(OrdRef+1+1,pass(OrdRef),i,:) = track(m-i,:);
430                RF(OrdRef+1+1,pass(OrdRef),i,:) = trackface(m-i,:);
431            end
432%             pass(OrdRef) = pass(OrdRef) + 1;
433            count = count + 1;             
434        end 
435    end
436    n = n+1;  % path calculation of current point end,  go to next point on the same level
437   
438    
439    end
440    b = 1;  % go back to part 1
441  
442end
443
444
445
446for i = 1:OrdRef
447if (Reflex == 0)
448    if (rvsds == 0)
449        NbrSlns(i) = 0;
450    else 
451        NbrDSSlns(i) = 0;
452    end 
453end 
454if ((count ~= 0) || (oldReflex == 1) || (Reflex) == 1)
455    Reflex = 1;
456else 
457    if (rvsds == 0)
458        NbrSlns(i) = 0;
459    else 
460        NbrDSSlns(i) = 0;
461    end
462end 
463    
464end
