1function blk = Blocking(P1_re,P2_re)
 2% Check if the two points are blocked
 3
 4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
 5% History:  2011.9.20
 6
 7% Updated by: Xuhong Li <xuhong@ftw.at>
 8% History: 2013.08.01
 9
10coder.extrinsic('exit')
11coder.extrinsic('disp')
12
13global vnfaces Nblock MSKFlag CEP wedges;
14
15n = zeros(3,1);
16face = zeros(2,1);
17
18blk = 0;
19i = 1;
20P1 = reshape(P1_re,3,1);
21P2 = reshape(P2_re,3,1);
22dp = P2-P1;
23
24
25while (i <= Nblock) && (blk ~= 1)
26    j = 1;
27    while (j <= 6) && (blk ~= 1)
28        face(1) = i;
29        face(2) = j;
30        n(:,1) = vnfaces(i,j,:);
31        cdpr = dp'*n;
32        if (abs(cdpr) > 1.0e-3)
33            x1 = wedges(i,j,1,1);
34            y1 = wedges(i,j,1,2);
35            z1 = wedges(i,j,1,3);
36            x2 = wedges(i,j,2,1);
37            y2 = wedges(i,j,2,2);
38            z2 = wedges(i,j,2,3); 
39            x0 = wedges(i,j,3,1);
40            y0 = wedges(i,j,3,2);
41            z0 = wedges(i,j,3,3);
42            dtr = (P1(1)-P2(1))*((y1-y0)*(z2-z0)-(y2-y0)*(z1-z0))-...
43                  (P1(2)-P2(2))*((x1-x0)*(z2-z0)-(x2-x0)*(z1-z0))+...
44                  (P1(3)-P2(3))*((x1-x0)*(y2-y0)-(x2-x0)*(y1-y0));
45            if (dtr == 0)
46                disp('Error: the determinant is equal to zero in bloking function!');
47                exit; 
48            else 
49                t = ((P1(1)-x0)*((y1-y0)*(z2-z0)-(y2-y0)*(z1-z0))+...
50                    (P1(2)-y0)*((x2-x0)*(z1-z0)-(x1-x0)*(z2-z0))+...
51                    (P1(3)-z0)*((x1-x0)*(y2-y0)-(x2-x0)*(y1-y0)))/dtr;
52                if (t > (-1.0e-6)) && (t < (1+(1.0e-6)))
53                    ip = P2 - P1;
54                    ip = P1 + t*ip;
55                    af = AppFace(ip,face);
56                    if (af == 1)
57                        if (MSKFlag(i) == 1) || (CEP(i) == 1)
58                            % masking if the block does not allow penetration or if it is a PEC
59                            if (abs(t)>1.0e-6) && (abs(t-1)>1.0e-6)
60                                blk = 1;
61                            end
62                        end 
63                    end
64                end 
65            end 
66        end 
67        j = j + 1;
68    end 
69    i = i + 1;
70end
