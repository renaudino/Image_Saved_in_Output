1function EffConstruct_Image(OrdRef,tx,rvsds)
  2% Data construction
  3
  4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
  5% History: 2011.9.21
  6
  7% Updated by: Xuhong Li <xuhong@ftw.at>
  8% History: 2013.08.01
  9% Combine Construct and CalcPath, calculate valid reflection path without
 10% build whole image tree and using recursion: EffConstruct, EffPointsImages
 11% Mpath: 1~2 ----- face,  3~5 ----- point
 12      
 13
 14global vnfaces recept Nblock Reflex NbrSlns NbrDSSlns oldReflex PSLN FSLN ISLN RP RF Mpath wedges;
 15
 16x = 0;
 17app = 0;
 18cache = 0;
 19tempvalue = 0;
 20count = 0;
 21oldReflex = Reflex;
 22
 23%--------------- unify variable size for code convertion -----------------%
 24% pass = ones(OrdRef,1);
 25pass = zeros(OrdRef,1);
 26leveltemp = zeros(1,Nblock*6,5);
 27pointVcheck = zeros(3,1);
 28extremcheck = zeros(3,1);
 29pointcheck = zeros(3,1);
 30facecheck = zeros(2,1);
 31Temp3 = zeros(3,1);
 32
 33track = zeros(OrdRef+1,3);
 34trackface = zeros(OrdRef+1,2);
 35trackimage = zeros(OrdRef+1,3);
 36vnfaces_re = zeros(3,1);
 37
 38pointV = zeros(3,1);
 39extrem = zeros(3,1);
 40% back = zeros(1,3);
 41fface = zeros(2,1);
 42numberuse = zeros(1,Nblock*6);
 43loopdomain = zeros(OrdRef+1,1);
 44%-------------------------------------------------------------------------%
 45
 46loopdomain(1) = 1;  % only Tx in the first level
 47
 48for i = 1:OrdRef   % build up the original image tree
 49    loopdomain(i+1) = EffPointsImages(i);  
 50end
 51
 52extrem(:,1) = recept;
 53b = 1;  % flag
 54golevel = 2;
 55goflag = 0; % original image tree: 0    updated image tree: 1
 56jumplevel = OrdRef+1; % pointer. from which point to update the image tree
 57order = 1;
 58
 59while (jumplevel > 1 && order == 1) % update the image tree until go back to Tx
 60    %% part 1
 61    % 1. Update the image tree until there are useful imagepoints in the last
 62    %    level, then go to part 2
 63    % 2. Calculate valid lower order (smaller than 'OrdRef') reflection path after updating image tree
 64    
 65    if OrdRef == 1  % if reflection order is 1, set order to 0, in order to jump out the loop
 66        order = 0;
 67    end
 68    
 69    while (b ==1)
 70        p = 0;
 71        while (golevel ~= OrdRef+1 && golevel > 1)
 72            
 73            if goflag ==1
 74                
 75                leveltemp = Mpath(golevel,2:end,:);  % remove the first used point, renew the tree from the second point on this level
 76                Mpath(golevel,:,:) = 0;
 77                Mpath(golevel,1:(Nblock*6-1),:) = leveltemp;
 78                
 79                Mpath(golevel+1:end,:,:) = 0;        % clear the last few levels for new imagepoints
 80                loopdomain(golevel+1:end) = 0;
 81                
 82                for m = golevel+1:OrdRef+1
 83                    loopdomain(m) = EffPointsImages(m-1);
 84                end
 85                
 86                loopdomain(golevel) = loopdomain(golevel)-1;
 87                
 88            end
 89            
 90            
 91            for n = golevel:OrdRef   % backtrack from the new point
 92                glob = 1;
 93                j = n;
 94                back = Mpath(n,1,:);
 95                extrem(:,1) = recept;
 96                m = 1;
 97                
 98                while (j ~=1 && glob == 1)
 99                    bt1 = reshape(back(1:2),1,2);    %  face
100                    bt2 = reshape(back(3:5),3,1);    % point
101                    bt3 = reshape(extrem,3,1);
102                    pointV(:,1) = Breakthrough(bt1,bt2,bt3);
103                    
104                    track(m,:) = pointV;    % coordinates of reflection point
105                    trackface(m,:) = back(1:2);    % face
106                    trackimage(m,:) = back(3:5);   % image
107                    difftest = extrem - pointV;
108                    
109%                     if (back(1) ~= 0)
110                        vnfaces_re(:,1) = vnfaces(trackface(m,1),trackface(m,2),:);
111                        
112                        if ((vnfaces_re'*difftest) <= 0)
113                            glob = 0;
114                            break;
115                        end
116%                     end
117                    
118%                     if (back(1) == 0)
119%                         app = AppGround(pointV);
120%                     else
121                        back_temp=back(1,1:2);
122                        back_temp=back_temp';
123                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
124%                         app = AppFace(pointV,back_temp); % Attention AppFace input dimension match
125                        app = AppFace2(pointV,squeeze(wedges(back_temp(1),back_temp(2),:,:)).');
126                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
127%                     end
128                    
129                    if (app == 1)
130                        cache = Blocking(extrem,pointV);
131                        if (cache == 1)
132                            glob = 0;
133                            break;
134                        end
135                        
136                        fface(1) = back(1);   % index of the current block
137                        
138                        if (fface(1) > 0)
139                            
140                            for Nface = 1:6
141                                if (Nface ~= back(2))
142                                    fface(2) = Nface;
143                                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
144%                                     appf = AppFace(pointV,fface);
145                                    appf = AppFace2(pointV,squeeze(wedges(fface(1),fface(2),:,:)).');
146                                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
147                                    
148                                    if (appf == 1)
149                                        glob = 0;
150                                        break;
151                                    end
152                                end
153                            end
154                            tempvalue = norm(extrem - pointV);  % test
155                            if (tempvalue < 1.0e-12)
156                                
157                                glob = 0;
158                                break;
159                            end
160                        end
161                        
162                    else
163                        glob = 0;
164                        break;
165                    end
166                    
167                    extrem(:,1) = pointV;
168                    j = j-1;
169                    back = Mpath(j,1,:); % the first point of upper level
170                    m = m+1;    %  the reflection point is okay, go to the upper level
171                    
172                end    % while (j ~=0)
173                
174                
175                if (glob == 1)
176                    Temp3(:,1)=tx;
177                    cache = Blocking(Temp3,pointV);   %  input of Blocking should be 3*1, so use tx'
178                    
179                    if (cache == 1)
180                        glob = 0;
181                    end
182                end
183                if (glob == 1)
184                    Reflex = 1;
185                    if (rvsds == 0)
186                        pass(n-1) = pass(n-1) + 1; % test
187                        NbrSlns(n-1) = pass(n-1);
188                        for i = 1:(m-1)
189                            PSLN(n-1,pass(n-1),i,:) = track(m-i,:);
190                            FSLN(n-1,pass(n-1),i,:) = trackface(m-i,:);
191                            ISLN(n-1,pass(n-1),i,:) = trackimage(m-i,:);
192                        end
193                        %pass(n-1) = pass(n-1) + 1;
194                        count = count + 1;
195                    else
196                        pass(n-1) = pass(n-1) + 1; % test
197                        NbrDSSlns(n) = pass(n-1);
198                        
199                        for i = 1:(m-1)
200                            
201                            RP(n,pass(n-1),i,:) = track(m-i,:);
202                            RF(n,pass(n-1),i,:) = trackface(m-i,:);
203                        end
204                        %                     pass(n-1) = pass(n-1) + 1;
205                        count = count + 1;
206                    end
207                end
208                
209                
210            end
211            
212            
213            if goflag == 0
214                
215                golevel = OrdRef+1;
216                goflag = 1;
217                
218            elseif goflag == 1
219                
220                if loopdomain(OrdRef) == 1 % special case: only one point left on the last second level
221                    p = 1;  % flag
222                    jumplevel = OrdRef-1;
223                    pp = 0;
224                    
225                    while jumplevel >= 1 && pp == 0 %  backtrack until there is more than one valid point on that level,
226                        
227                        if ( loopdomain(jumplevel)==1 )  % important for the next start point
228                            jumplevel = jumplevel-1;
229                            pp = 0;
230                        else
231                            pp = 1;
232                        end
233                    end
234                    
235                    golevel = OrdRef+1;
236                    
237                else
238                    
239                    p = 0;
240                    golevel = OrdRef+1;
241                end
242                
243            end
244        end
245        
246        
247        
248        %------check if the point on the last level is useful------%
249        if golevel == OrdRef+1
250            
251            x = 0;
252            for iii =1:loopdomain(golevel)
253                facecheck(:,1) = Mpath(golevel,iii,1:2);    % 1*2
254                pointcheck(:,1) = Mpath(golevel,iii,3:5);   % 3*1
255                extremcheck(:,1) = reshape(recept,3,1); %3*1
256                pointVcheck(:,1) = Breakthrough(facecheck',pointcheck,extremcheck);
257                
258%                 if (facecheck(1) == 0)   % if ground
259%                     app = AppGround(pointVcheck);
260%                 else
261                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
262%                     app = AppFace(pointVcheck,facecheck); % Attention AppFace input dimension match
263                    app = AppFace2(pointVcheck,squeeze(wedges(facecheck(1),facecheck(2),:,:)).');
264                    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
265%                 end
266                
267                if app == 1
268                    x = x+1;
269                    numberuse(x) = iii;   %  the useful imagepoints index on the last level
270                end
271            end
272
273            if x == 0   %  if no useful imagepoints
274                if p == 1  % update the image tree from the 1st point on jumplevel
275                    golevel = jumplevel;
276                elseif p == 0
277                    golevel = golevel-1;
278                end
279                b = 1; % there is no valid point on the last level, continue part 1
280            elseif x >= 1 && x <loopdomain(golevel)    %   renew the OrdRef+1 level
281                for u = 1:x
282                    leveltemp(1,u,:) = Mpath(golevel,numberuse(u),:);
283                end
284                Mpath(golevel,:,:) = 0;
285                Mpath(golevel,1:x,:) = leveltemp(1,1:x,:);
286                if p == 1  % update the image tree from the 1st point on jumplevel
287                    golevel = jumplevel;
288                elseif p == 0
289                    golevel = golevel-1;    
290                end
291                b = 0; % there are valid points on the last level, go to part 2
292            elseif x == loopdomain(golevel)  %  all imagepoints can be used
293                if p == 1  % update the image tree from the 1st point on jumplevel
294                    golevel = jumplevel;
295                elseif p == 0
296                    golevel = golevel-1;   
297                end
298                b = 0;  % there are valid points on the last level, go to part 2  
299            end   
300        else
301            b = 0;
302        end       
303    end
304    
305    loopdomain(OrdRef+1) = x; %  # valid points on the last level
306    
307    
308    %% part 2
309    % Backtrack from the points on the last level, calculate full order ('OrdRef') reflection path
310    
311    
312    n = 1;
313    while (n <= loopdomain(OrdRef+1) )   % loop for useful image points in the last level
314        glob = 1;
315        j = OrdRef+1;
316        back = Mpath(OrdRef+1,n,:);
317        m = 1;
318        extrem(:,1) = recept;
319        
320        while (j ~=1 && glob == 1)
321            bt1 = reshape(back(1:2),1,2);    % face
322            bt2 = reshape(back(3:5),3,1);    % point
323            bt3 = reshape(extrem,3,1);
324            pointV(:,1) = Breakthrough(bt1,bt2,bt3);
325            
326            track(m,:) = pointV;            % coordinates of reflection point
327            trackface(m,:) = back(1:2);     % face
328            trackimage(m,:) = back(3:5);    % image
329            difftest = extrem - pointV;
330            
331%             if (back(1) ~= 0)
332                vnfaces_re(:,1) = vnfaces(trackface(m,1),trackface(m,2),:);
333                
334                if ((vnfaces_re'*difftest) <= 0)
335                    glob = 0;
336                    break;
337                end
338%             end
339%             if (back(1) == 0) 
340%                 app = AppGround(pointV);
341%             else
342                back_temp=back(1,1:2);
343                back_temp=back_temp';
344                %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
345%                 app = AppFace(pointV,back_temp); % Attention AppFace input dimension match
346                app = AppFace2(pointV,squeeze(wedges(back_temp(1),back_temp(2),:,:)).');
347                %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
348                
349%             end
350            
351            if (app == 1)
352                cache = Blocking(extrem,pointV);
353                if (cache == 1)
354                    glob = 0;
355                    break;
356                end
357                
358                fface(1) = back(1);   % index of the current block
359                if (fface(1) > 0)
360                    for Nface = 1:6
361                        if (Nface ~= back(2))
362                            fface(2) = Nface;
363                            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
364%                             appf = AppFace(pointV,fface);
365                            appf = AppFace2(pointV,squeeze(wedges(fface(1),fface(2),:,:)).');
366                            %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
367                            if (appf == 1)
368                                glob = 0;
369                                break;
370                            end
371                        end
372                    end
373                    
374                    tempvalue = norm(extrem - pointV);
375                    if (tempvalue < 1.0e-12)
376                        glob = 0;
377                        break;
378                    end
379                end
380            else
381                glob = 0;
382                break;
383            end
384            
385            extrem(:,1) = pointV;
386            j = j-1;
387            back = Mpath(j,1,:);
388            m = m+1;
389            
390        end
391        
392        if (glob == 1)
393            Temp3(:,1)=tx;
394            cache = Blocking(Temp3,pointV);
395            
396            if (cache == 1)
397                glob = 0;
398            end
399        end
400        if (glob == 1)
401            Reflex = 1;
402            if (rvsds == 0)
403                pass(OrdRef) = pass(OrdRef) + 1; % test
404                NbrSlns(OrdRef) = pass(OrdRef);
405                for i = 1:(m-1)
406                    PSLN(OrdRef,pass(OrdRef),i,:) = track(m-i,:);
407                    FSLN(OrdRef,pass(OrdRef),i,:) = trackface(m-i,:);
408                    ISLN(OrdRef,pass(OrdRef),i,:) = trackimage(m-i,:);
409                end
410                %             pass(OrdRef) = pass(OrdRef) + 1;
411                count = count + 1;
412            else
413                pass(OrdRef) = pass(OrdRef) + 1; %test
414                NbrDSSlns(OrdRef+1+1) = pass(OrdRef);
415                for i = 1:(m-1)
416                    RP(OrdRef+1+1,pass(OrdRef),i,:) = track(m-i,:);
417                    RF(OrdRef+1+1,pass(OrdRef),i,:) = trackface(m-i,:);
418                end
419                %             pass(OrdRef) = pass(OrdRef) + 1;
420                count = count + 1;
421            end
422        end
423        n = n+1;  % path calculation of current point end,  go to next point on the same level    
424    end
425    b = 1;  % go back to part 1
426end
427
428
429
430for i = 1:OrdRef
431    if (Reflex == 0)
432        if (rvsds == 0)
433            NbrSlns(i) = 0;
434        else
435            NbrDSSlns(i) = 0;
436        end
437    end
438    if ((count ~= 0) || (oldReflex == 1) || (Reflex) == 1)
439        Reflex = 1;
440    else
441        if (rvsds == 0)
442            NbrSlns(i) = 0;
443        else
444            NbrDSSlns(i) = 0;
445        end
446    end
447    
448end
