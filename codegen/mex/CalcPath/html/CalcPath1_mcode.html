1function CalcPath(tx,rx,ordmax,rvsds)
  2% Calculates the path followed by a ray, i.e. the reflection points
  3
  4% Authors: Francesco Mani <francesco.mani@uclouvain.be> and Mingming Gan <gan@ftw.at>
  5% History: 2011.9.26
  6%          2011.10.4 updated after learning the diffuse scattering
  7
  8% Updated by: Xuhong Li <xuhong@ftw.at>
  9% History: 2013.08.01
 10
 11global vnfaces NbrSlns PSLN FSLN Reflex NbrDSSlns RP RF oldReflex Mtrun Nbrimg wedges;
 12
 13ornum = 1;
 14% pass = 1;
 15pass = 0; % test
 16count = 0;
 17maxnumber = max(Nbrimg(:));
 18
 19app = 0;
 20
 21%--------------- unify variable size for code convertion -----------------%
 22fface = zeros(2,1);
 23curs = zeros(1,9);
 24right = zeros(1,9);
 25back = zeros(1,9);
 26extrem = zeros(3,1);
 27pointV = zeros(3,1);
 28vnfaces_re = zeros(3,1);
 29lastface = zeros(2,1);
 30track = zeros(ordmax,3);
 31trackface = zeros(ordmax,2);
 32Temp3=zeros(3,1);
 33%-------------------------------------------------------------------------%
 34
 35
 36curs(1,:) = Mtrun((ordmax+1),ornum,:);
 37right = curs; 
 38oldReflex = Reflex;
 39
 40while (right(1)~=0) % if not empty, then continue
 41    back = right;
 42    extrem(:,1) = rx;
 43    glob = 1;
 44    m = 1;
 45   while (back(8)~=0 && glob ==1) 
 46       bt1 = reshape(back(2:3),1,2);
 47       bt2 = reshape(back(4:6),3,1);
 48       bt3 = reshape(extrem,3,1);
 49       pointV(:,1) = Breakthrough(bt1,bt2,bt3);
 50       track(m,:) = pointV;   
 51       trackface(m,:) = back(2:3);
 52       difftest = extrem - pointV;
 53       
 54%        if (back(2) ~= 0)
 55           vnfaces_re(:,1) = vnfaces(trackface(m,1),trackface(m,2),:);
 56           if ((vnfaces_re'*difftest) <= 0)
 57               glob = 0;
 58               break;
 59           end
 60%        end
 61       
 62%        if (back(2) == 0)
 63%            app = AppGround(pointV);
 64%        else
 65           back_temp=back(1,2:3);
 66           back_temp=back_temp';
 67           %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 68%            app = AppFace(pointV,back_temp);
 69           app = AppFace2(pointV,squeeze(wedges(back_temp(1),back_temp(2),:,:)).');
 70           %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 71%        end
 72       
 73       if (app == 1)
 74           cache = Blocking(extrem,pointV);
 75           if (cache == 1)
 76               glob = 0;
 77               break;
 78           end
 79           
 80           fface(1) = back(2);
 81%            if (fface(1) > 0)
 82               for j = 1:6
 83                    if (j ~= back(3))
 84                        fface(2) = j;
 85                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 86%                         appf = AppFace(pointV,fface);
 87                        appf = AppFace2(pointV,squeeze(wedges(fface(1),fface(2),:,:)).');
 88                        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 89                        if (appf == 1)
 90                            glob = 0;
 91                            break;
 92                        end
 93                    end 
 94                end
 95                tempvalue = norm(extrem - pointV);
 96            
 97                if (tempvalue < 1.0e-12)
 98                    glob = 0;
 99                    break;
100                end 
101%             end 
102        else 
103            glob = 0;
104            break;
105        end
106    
107        extrem(:,1) = pointV;
108        lastface(:,1) = back(2:3);
109        back = reshape(Mtrun(back(8),back(9),:),1,9);
110        m = m+1;    % means the reflection point found is okay, then go to the upper level
111
112
113   end 
114    
115    lastface(:,1) = back(2:3);
116    if (glob == 1)
117        
118        Temp3(:,1)=tx;
119        cache = Blocking(Temp3,pointV);   %  input of Blocking should be 3*1, so use tx'
120        
121        if (cache == 1)
122            glob = 0;
123        end 
124    end
125    if (glob == 1)
126        Reflex = 1;
127        if (rvsds == 0)
128            pass = pass + 1;
129            NbrSlns(ordmax) = pass;
130            for i = 1:(m-1)
131                PSLN(ordmax,pass,i,:) = track(m-i,:);
132                FSLN(ordmax,pass,i,:) = trackface(m-i,:);
133            end
134%             pass = pass + 1;
135            count = count + 1;
136        else 
137            pass = pass + 1;
138            NbrDSSlns(ordmax+1) = pass;
139            for i = 1:(m-1)
140                RP(ordmax+1,pass,i,:) = track(m-i,:);
141                RF(ordmax+1,pass,i,:) = trackface(m-i,:);
142            end
143%             pass = pass + 1;
144            count = count + 1;             
145        end 
146    end
147    ornum = ornum+1;
148    
149    if ornum <= maxnumber
150         right = reshape(Mtrun((ordmax+1),ornum,:),1,9);
151    else
152       break;
153    end
154end 
155if (Reflex == 0)
156    if (rvsds == 0)
157        NbrSlns(ordmax) = 0;
158    else 
159        NbrDSSlns(ordmax) = 0;
160    end 
161end 
162if ((count ~= 0) || (oldReflex == 1) || (Reflex) == 1)
163    Reflex = 1;
164else 
165    if (rvsds == 0)
166        NbrSlns(ordmax) = 0;
167    else 
168        NbrDSSlns(ordmax) = 0;
169    end
170end
